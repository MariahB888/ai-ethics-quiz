import javax.swing.*; import java.awt.*; import java.awt.event.ActionEvent; import java.awt.event.ActionListener; import java.util.HashMap; import java.util.Map; public class AIQuizTree { private static JFrame frame; private static JPanel panel; private static JLabel questionLabel; private static JButton option1, option2, option3; private static String currentQuestion; private static Map questionTree = new HashMap<>(); static { questionTree.put("Start", new String[]{"Should AI be allowed to make medical decisions?", "Yes", "No", "Depends"}); questionTree.put("Yes", new String[]{"Should AI replace doctors?", "Yes, it's more accurate", "No, doctors are needed", "Only for diagnosis"}); questionTree.put("No", new String[]{"Should AI assist in medical decisions?", "Yes, as an advisor", "No, too risky", "Only in emergencies"}); questionTree.put("Depends", new String[]{"What factors should decide?", "Data accuracy", "Human oversight", "Case-by-case"}); questionTree.put("Yes, it's more accurate", new String[]{"Who should oversee AI medical systems?", "Doctors", "Tech companies", "The government"}); questionTree.put("No, doctors are needed", new String[]{"Should AI have any role in medicine?", "Yes, but limited", "No role at all", "Only for research"}); questionTree.put("Only for diagnosis", new String[]{"What about AI in mental health?", "Helpful for therapy", "Not reliable", "Only as a tool"}); } public static void main(String[] args) { frame = new JFrame("AI Ethics Decision Tree"); frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); frame.setSize(700, 500); panel = new JPanel(); panel.setLayout(new BorderLayout()); panel.setBackground(new Color(30, 60, 114)); questionLabel = new JLabel("Welcome to the AI Ethics Tree!", SwingConstants.CENTER); questionLabel.setForeground(Color.WHITE); questionLabel.setFont(new Font("Arial", Font.BOLD, 18)); panel.add(questionLabel, BorderLayout.NORTH); JPanel buttonPanel = new JPanel(); buttonPanel.setLayout(new GridLayout(3, 1, 10, 10)); buttonPanel.setBackground(new Color(30, 60, 114)); option1 = new JButton(); option2 = new JButton(); option3 = new JButton(); buttonPanel.add(option1); buttonPanel.add(option2); buttonPanel.add(option3); panel.add(buttonPanel, BorderLayout.CENTER); option1.addActionListener(new ButtonClickListener()); option2.addActionListener(new ButtonClickListener()); option3.addActionListener(new ButtonClickListener()); frame.add(panel); frame.setVisible(true); currentQuestion = "Start"; loadQuestion(); } private static void loadQuestion() { if (questionTree.containsKey(currentQuestion)) { String[] data = questionTree.get(currentQuestion); questionLabel.setText(data[0]); option1.setText(data[1]); option2.setText(data[2]); option3.setText(data[3]); option1.setVisible(true); option2.setVisible(true); option3.setVisible(true); } else { questionLabel.setText("End of Path: " + currentQuestion); option1.setVisible(false); option2.setVisible(false); option3.setVisible(false); } } static class ButtonClickListener implements ActionListener { public void actionPerformed(ActionEvent e) { JButton source = (JButton) e.getSource(); currentQuestion = source.getText(); loadQuestion(); } } }
